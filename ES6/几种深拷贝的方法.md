4、for...in
```
function extend(target){
  let sources = Array.prototype.slice.call(arguments, 1);
  for (let i = 0; i < sources.length; i += 1) {
    let source = sources[i];
    for (let key in source) {
      if (source.hasOwnProperty(key)) {
        target[key] = source[key];
      }
    }
  }
  return target;
}
extend({}, obj1, obj2);
```
